Potential Reentrancy Vulnerability in `deployPool()` Function

The `deployPool()` function in the `StakingPoolFactory` contract transfers the platform fee to the `platformAddress` after deploying the new staking pool contract. This could potentially lead to a reentrancy vulnerability if the `platformAddress` is a malicious contract that can call back into the `StakingPoolFactory` contract.

           user.rewards = earned(poolId, account);

       require(tokenAddress != address(0), "Zero address");
       
       try IERC20Metadata(tokenAddress).decimals() returns (uint8 decimals) {
           // Check if token has transfer and transferFrom functions
           try IERC20(tokenAddress).transfer(address(this), 0) {} catch {
               emit TokenValidationFailed(tokenAddress, "Transfer function check failed");
               revert("Invalid token: transfer");
           }
           
           try IERC20(tokenAddress).transferFrom(address(this), address(this), 0) {} catch {
               emit TokenValidationFailed(tokenAddress, "TransferFrom function check failed");
               revert("Invalid token: transferFrom");
           }

           return decimals;
       } catch {
           emit TokenValidationFailed(tokenAddress, "Decimals check failed");
           revert("Invalid token: decimals");
       }

       address _rewardToken,
       uint256 _cooldownPeriod,
       uint256 _lockPeriod
   ) external onlyOwner returns (uint256 poolId) {
       require(_cooldownPeriod >= MIN_COOLDOWN_PERIOD, "Cooldown too short");
       require(_lockPeriod >= MIN_LOCK_PERIOD, "Lock period too short");

       // Validate tokens and get decimals
       uint8 stakingDecimals = validateToken(_stakingToken);
       uint8 rewardDecimals = validateToken(_rewardToken);

       poolId = poolCount++;
       PoolInfo storage pool = pools[poolId];
       
       pool.stakingToken = IERC20(_stakingToken);
       pool.rewardToken = IERC20(_rewardToken);
       pool.rewardsDuration = 30 days;
       pool.cooldownPeriod = _cooldownPeriod;
       pool.lockPeriod = _lockPeriod;
       pool.stakingTokenDecimals = stakingDecimals;
       pool.rewardTokenDecimals = rewardDecimals;
       pool.isPaused = false;

       emit PoolCreated(poolId, _stakingToken, _rewardToken, stakingDecimals, rewardDecimals);
   }

   /*//////////////////////////////////////////////////////////////
                             REWARD CALCULATIONS
   //////////////////////////////////////////////////////////////*/

   function lastTimeRewardApplicable(uint256 poolId) public view validatePoolId(poolId) returns (uint256) {
       return block.timestamp < pools[poolId].periodFinish ? block.timestamp : pools[poolId].periodFinish;
   }

   function rewardPerToken(uint256 poolId) public view validatePoolId(poolId) returns (uint256) {
       if (pools[poolId].totalStaked == 0) {
           return pools[poolId].rewardPerTokenStored;
       }

       uint256 timeElapsed = lastTimeRewardApplicable(poolId).sub(pools[poolId].lastUpdateTime);
       uint256 currentRewardRate = getCurrentRewardRate(poolId);
       
       return pools[poolId].rewardPerTokenStored.add(
           timeElapsed.mul(currentRewardRate).mul(1e18).div(pools[poolId].totalStaked)
       );
   }

   function getCurrentRewardRate(uint256 poolId) public view validatePoolId(poolId) returns (uint256) {
       if (block.timestamp >= pools[poolId].periodFinish) return 0;
       return pools[poolId].totalRewardPool.div(pools[poolId].rewardsDuration);
   }

   function earned(uint256 poolId, address account) public view validatePoolId(poolId) returns (uint256) {
       UserInfo storage user = userInfo[poolId][account];
       
       return user.stakeBalance.mul(
           rewardPerToken(poolId).sub(user.rewardPerTokenPaid)
       ).div(1e18).add(user.rewards);
   }

       user.lastStakeTime = block.timestamp;

       require(pool.stakingToken.transferFrom(msg.sender, address(this), amount), "Transfer failed");
       uint256 balanceAfter = pool.stakingToken.balanceOf(address(this));
       require(balanceAfter.sub(balanceBefore) == amount, "Fee on transfer not supported");
       
       emit Staked(poolId, msg.sender, amount);
   }

       validatePoolId(poolId)
       checkLockPeriod(poolId)
       checkCooldown(poolId)

       pool.totalStaked = pool.totalStaked.sub(amount);
       user.stakeBalance = user.stakeBalance.sub(amount);

       require(pool.stakingToken.transfer(msg.sender, amount), "Transfer failed");
       uint256 balanceAfter = pool.stakingToken.balanceOf(msg.sender);

pragma solidity ^0.8.28;

import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/proxy/Clones.sol";

interface IStakingPool {
    function initialize(address owner) external;
}

contract StakingPoolFactory is Ownable, Pausable {
    using Clones for address;

    /*//////////////////////////////////////////////////////////////
                                 STORAGE
    //////////////////////////////////////////////////////////////*/

    address public immutable implementation;
    address[] public pools;
    mapping(address => bool) public isPoolFromFactory;
    
    // Platform fee storage
    address public platformAddress;
    uint256 public createPoolFee;

    /*//////////////////////////////////////////////////////////////
                                 EVENTS
    //////////////////////////////////////////////////////////////*/

    event PoolDeployed(address indexed pool, address indexed owner);
    event PlatformAddressUpdated(address indexed newPlatformAddress);
    event CreatePoolFeeUpdated(uint256 newFee);

    /*//////////////////////////////////////////////////////////////
                               CONSTRUCTOR
    //////////////////////////////////////////////////////////////*/

    constructor(address _implementation) Ownable(msg.sender) {
        require(_implementation != address(0), "Zero address");
        implementation = _implementation;
    }

    /*//////////////////////////////////////////////////////////////
                            EXTERNAL FUNCTIONS
    //////////////////////////////////////////////////////////////*/

    function deployPool() external payable whenNotPaused returns (address pool) {
        // Check if the sent ETH matches the required fee
        require(msg.value >= createPoolFee, "Insufficient fee");
        
        pool = implementation.clone();
        IStakingPool(pool).initialize(msg.sender);
        
        pools.push(pool);
        isPoolFromFactory[pool] = true;
        
        // Transfer the fee to platform address
        (bool success, ) = platformAddress.call{value: msg.value}("");
        require(success, "Fee transfer failed");
        
        emit PoolDeployed(pool, msg.sender);
    }

    function getAllPools() external view returns (address[] memory) {
        return pools;
    }

    function getPoolCount() external view returns (uint256) {
        return pools.length;
    }
