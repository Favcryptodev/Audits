Key Findings

Medium
M-01

Potential Denial of Service (DoS) through Excessive Gas Consumption

The `createCollection` function does not have a limit on the number of collections that can be created. This could potentially lead to a Denial of Service (DoS) attack if an attacker creates an excessive number of collections, consuming a large amount of gas and slowing down the entire system.
LOCATION
CustomERC721Factory.sol:createCollection
IMPACT
An attacker could create an unlimited number of collections, consuming a large amount of gas and slowing down the entire system, effectively denying service to other users.
RECOMMENDATION
Consider adding a limit on the number of collections that can be created, either by the entire contract or per user. Alternatively, you could implement a fee structure that increases with the number of collections created to disincentivize excessive creation.
Low
L-01

Lack of Input Validation

The `createCollection` function does not perform any input validation on the parameters passed to it. This could potentially lead to issues if the input is not properly sanitized.
LOCATION
CustomERC721Factory.sol:createCollection
IMPACT
Improper input handling could lead to issues such as unintended behavior or potential security vulnerabilities.
RECOMMENDATION
Implement input validation on the parameters passed to the `createCollection` function to ensure that the input is within the expected range and format.
Informational
I-01

Potential Gas Optimization

The `totalCollections` function could be optimized by caching the length of the `allCollections` array instead of calculating it every time the function is called.
LOCATION
CustomERC721Factory.sol:totalCollections
IMPACT
Repeated calls to the `totalCollections` function could lead to unnecessary gas consumption.
RECOMMENDATION
Consider caching the length of the `allCollections` array and updating it whenever a new collection is added or removed.

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import "@openzeppelin/contracts/proxy/Clones.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "./CustomERC721.sol";

/**
 * @title CustomERC721Factory
 * @dev Factory contract for deploying minimal proxy clones of the CustomERC721 contract
 */
contract CustomERC721Factory is Ownable, ReentrancyGuard {
    // The address of the implementation contract to clone
    address payable public immutable implementation;
    
    // Platform address to receive fees
    address payable public platformAddress;
    
    // Mapping to track all created collections
    mapping(address => bool) public isCollectionFromFactory;
    
    // Array to store all created collection addresses
    address[] public allCollections;
    
    // Mapping to track collections created by a specific address
    mapping(address => address[]) public collectionsByCreator;

    event CollectionCreated(
        address indexed creator,
        address indexed collectionAddress,
        string name,
        string symbol,
        uint256 maxSupply,
        uint256 mintFee,
        uint256 creatorFee,
        uint256 mintStartTime
    );
    
    event PlatformAddressUpdated(address oldAddress, address newAddress);

    /**
     * @dev Constructor for the factory
     * @param _implementation Address of the implementation contract
     * @param _platformAddress Address to receive platform fees
     */
    constructor(
        address payable _implementation,
        address payable _platformAddress
    ) Ownable(msg.sender) {
        require(_implementation != address(0), "Implementation cannot be zero address");
        require(_platformAddress != address(0), "Platform address cannot be zero address");
        
        implementation = _implementation;
        platformAddress = _platformAddress;
    }

    /**
     * @dev Creates a new NFT collection by cloning the implementation contract
     * @param _name Name of the NFT collection
     * @param _symbol Symbol of the NFT collection
     * @param _baseURI Base URI for the token metadata
     * @param _maxSupply Maximum supply of tokens (0 for unlimited)
     * @param _mintFee Fee to mint a token
     * @param _creatorFee Royalty percentage (in basis points, 100 = 1%)
     * @param _mintStartTime Time when minting is allowed to start
     * @return newCollection Address of the newly created collection
     */
    function createCollection(
        string memory _name,
        string memory _symbol,
        string memory _baseURI,
        uint256 _maxSupply,
        uint256 _mintFee,
        uint256 _creatorFee,
        uint256 _mintStartTime
    ) external nonReentrant returns (address payable newCollection) {
        // Create clone
        newCollection = payable(Clones.clone(implementation));
        
        // Initialize the clone
        CustomERC721(newCollection).initialize(
            _name,
            _symbol,
            _baseURI,
            _maxSupply,
            _mintFee,
            _creatorFee,
            _mintStartTime,
            msg.sender
        );
        
        // Transfer ownership to the creator
        CustomERC721(newCollection).transferOwnership(msg.sender);
        
        // Track the collection
        isCollectionFromFactory[newCollection] = true;
        allCollections.push(newCollection);
        collectionsByCreator[msg.sender].push(newCollection);
        
        emit CollectionCreated(
            msg.sender,
            newCollection,
            _name,
            _symbol,
            _maxSupply,
            _mintFee,
            _creatorFee,
            _mintStartTime
        );
        
        return newCollection;
    }

    /**
     * @dev Updates the platform address
     * @param _newPlatformAddress New platform address
     */
    function setPlatformAddress(address payable _newPlatformAddress) external onlyOwner {
        require(_newPlatformAddress != address(0), "Platform address cannot be zero address");
        address oldAddress = platformAddress;
        platformAddress = _newPlatformAddress;
        emit PlatformAddressUpdated(oldAddress, _newPlatformAddress);
    }

    /**
     * @dev Returns total number of collections created
     */
    function totalCollections() external view returns (uint256) {
        return allCollections.length;
    }

    /**
     * @dev Returns all collections created by a specific address
     * @param _creator Creator address
     */
    function getCollectionsByCreator(address _creator) external view returns (address[] memory) {
        return collectionsByCreator[_creator];
    }

    /**
     * @dev Emergency withdrawal function for contract owner
     */
    function withdrawEther() external onlyOwner {
        uint256 balance = address(this).balance;
        require(balance > 0, "No balance to withdraw");
        (bool success, ) = payable(owner()).call{value: balance}("");
        require(success, "Withdrawal failed");
    }
}
