/**
 * Fetch recent buy transactions from Raydium pools
 * @returns {Promise<number>} - Total buy volume available to sell against
 */
async function fetchRecentBuyTransactions() {
    try {
        console.log("Fetching recent buy transactions from Raydium...");
        
        // Get pool transaction history
        const signatures = await connection.getSignaturesForAddress(
            new PublicKey(RAYDIUM_POOL_ADDRESS),
            { limit: 100 },
            'confirmed'
        );
        
        // If we've already checked transactions up to a certain point, only look at newer ones
        const newSignatures = recentBuys.lastCheckedSignature 
            ? signatures.filter(sig => sig.signature !== recentBuys.lastCheckedSignature).reverse()
            : signatures.reverse();
            
        if (newSignatures.length === 0) {
            console.log("No new transactions found");
            return 0;
        }
        
        // Update the most recent signature we've checked
        recentBuys.lastCheckedSignature = signatures[0].signature;
        
        let newBuyVolume = 0;
        
        // Process transactions in batches to avoid rate limits
        const BATCH_SIZE = 10;
        
        for (let i = 0; i < newSignatures.length; i += BATCH_SIZE) {
            const batch = newSignatures.slice(i, i + BATCH_SIZE);
            
            const transactionInfos = await Promise.all(
                batch.map(sig => connection.getTransaction(sig.signature, {
                    commitment: 'confirmed',
                    maxSupportedTransactionVersion: 0
                }))
            );
            
            for (const txInfo of transactionInfos) {
                if (!txInfo) continue;
                
                const isBuyTransaction = await identifyBuyTransaction(txInfo);
                
                if (isBuyTransaction) {
                    const buyAmount = calculateBuyAmount(txInfo);
                    
                    if (buyAmount > 0) {
                        newBuyVolume += buyAmount;
                        recentBuys.buyTransactions.push({
                            signature: txInfo.transaction.signatures[0],
                            amount: buyAmount,
                            timestamp: txInfo.blockTime,
                            processed: false
                        });
                        
                        console.log(`Found buy transaction: ${txInfo.transaction.signatures[0]} for ${buyAmount} tokens`);
                    }
                }
            }
            
            // Add delay between batches to avoid rate limits
            if (i + BATCH_SIZE < newSignatures.length) {
                await new Promise(resolve => setTimeout(resolve, 500));
            }
        }
        
        // Update the total buy volume
        recentBuys.totalBuyVolume += newBuyVolume;
        
        console.log(`Detected ${newBuyVolume} new buy volume. Total available: ${recentBuys.totalBuyVolume - recentBuys.processedSellVolume}`);
        
        return recentBuys.totalBuyVolume - recentBuys.processedSellVolume;
    } catch (error) {
        console.error("Error fetching buy transactions:", error);
        return 0;
    }
}

 * Identify if a transaction is a buy on Raydium
 * @param {TransactionResponse} txInfo - Transaction information

async function identifyBuyTransaction(txInfo) {
    try {

        const raydiumSwapProgramId = "675kPX9MHTjS2zt1qfr1NYHuzeLXfQM9H24wFSUt1Mp8";
        
        // Check if the transaction involves the Raydium swap program
        const involvesRaydiumSwap = txInfo.transaction.message.instructions.some(

        );
        
        if (!involvesRaydiumSwap) return false;

            return false; // Skip our own transactions

                    // 2. Our token is transferred to the user's wallet

/**
 * Calculate the buy amount from a transaction
 * @param {TransactionResponse} txInfo - Transaction information
 * @returns {number} - Buy amount
 */
function calculateBuyAmount(txInfo) {
    try {
        // Look for post token balances to determine token amounts
        if (txInfo.meta && txInfo.meta.postTokenBalances && txInfo.meta.preTokenBalances) {
            // Find our token in post balances
            const ourTokenPostBalances = txInfo.meta.postTokenBalances.filter(
                balance => balance.mint === mint
            );
            
            // Find our token in pre balances
            const ourTokenPreBalances = txInfo.meta.preTokenBalances.filter(
                balance => balance.mint === mint
            );
            
            // For each account that has our token
            for (const postBalance of ourTokenPostBalances) {
                // Find the corresponding pre-balance
                const preBalance = ourTokenPreBalances.find(
                    pre => pre.accountIndex === postBalance.accountIndex
                );
                
                // If we found a match and the balance increased, it's a buy
                if (preBalance) {
                    const preBal = Number(preBalance.uiTokenAmount.amount);
                    const postBal = Number(postBalance.uiTokenAmount.amount);
                    
                    if (postBal > preBal) {
                        // This account received tokens (bought)
                        return postBal - preBal;
                    }
                } else if (postBalance.uiTokenAmount.amount > 0) {
                    // New account with tokens (definitely a buy)
                    return Number(postBalance.uiTokenAmount.amount);
                }
            }
        }
        
        // Fallback - parse the logs to find transfer amounts
        if (txInfo.meta && txInfo.meta.logMessages) {
            for (const log of txInfo.meta.logMessages) {
                // Look for Transfer: X tokens to an account
                if (log.includes("Transfer") && log.includes("amount")) {
                    // Extract the amount - this is a simplified example
                    // In a real implementation, you'd need to parse the exact format
                    const match = log.match(/amount (\d+)/);
                    if (match && match[1]) {
                        return Number(match[1]);
                    }
                }
            }
        }
        
        return 0;
    } catch (error) {
        console.error("Error calculating buy amount:", error);
        return 0;
    }
}
